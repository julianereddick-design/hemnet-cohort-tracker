readJsonl is exported from lib/sold-transport.js, but that module throws at load unless
SCRAPE_FORCE_OXYLABS=1 is set (lib/sold-transport.js:22-26). To keep this persist script DB/IO-

```javascript
'use strict';
const fs = require('fs');
const { createClient } = require('../db');
const { upsertBooliSold, upsertHemnetSold } = require('../lib/sold-store');

function readJsonl(file) {
  return fs.readFileSync(file, 'utf8').split('\n').filter(Boolean).map((l) => JSON.parse(l));
}

function parseArgs(argv) {
  const out = { booli: null, hemnet: null, smoke: false };
  for (let i = 0; i < argv.length; i++) {
    if (argv[i] === '--booli') out.booli = argv[++i];
    else if (argv[i] === '--hemnet') out.hemnet = argv[++i];
    else if (argv[i] === '--smoke') out.smoke = true;
  }
  return out;
}

async function run(args) {
  const client = createClient();
  await client.connect();
  let booli = 0, hemnet = 0;
  try {
    if (args.booli) {
      for (const rec of readJsonl(args.booli)) { await upsertBooliSold(client, rec); booli++; }
    }
    if (args.hemnet) {
      for (const rec of readJsonl(args.hemnet)) { await upsertHemnetSold(client, rec); hemnet++; }
    }
  } finally {
    await client.end();
  }
  console.log(`persisted: booli=${booli} hemnet=${hemnet}`);
  return { booli, hemnet };
}

if (require.main === module) {
  const args = parseArgs(process.argv.slice(2));
  if (args.smoke) {
    const assert = require('assert');
    assert.strictEqual(typeof upsertBooliSold, 'function');
    assert.strictEqual(typeof upsertHemnetSold, 'function');
    const a = parseArgs(['--booli', 'x.jsonl', '--hemnet', 'y.jsonl']);
    assert.strictEqual(a.booli, 'x.jsonl');
    assert.strictEqual(a.hemnet, 'y.jsonl');
    console.log('smoke: ok');
    process.exit(0);
  }
  run(args).catch((err) => { console.error('Error:', err.message); process.exit(1); });
}

module.exports = { run, parseArgs };
```
Do NOT delete or modify the JSONL append in the fetchers — DB is store of record, JSONL stays
the raw landing/resume cache (D-04). This script is the one-way upsert pass JSONL → DB. Each
record loops through the store upsert, so re-running the same JSONL is idempotent (DB-03) via the
store's ON CONFLICT.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| JSONL file → persist script → Postgres | Parsed records (snake_case from Phase-15 parsers) cross into the DB; values must reach SQL only via bound parameters |
| caller client → store functions | The store receives a connected pg client; it never opens its own connection (keeps offline smoke DB-free) |
| .env → createClient (persist-sold.js) | DB credentials from environment only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-16-05 | Tampering (SQL injection) | every upsert in lib/sold-store.js | mitigate | All values pass through `$1,$2,…` bound parameters; column lists are static literals. Acceptance grep asserts no `${` interpolation in any query. JSONB evidence is bound as a parameter (JSON.stringify), never concatenated. |
| T-16-06 | Information disclosure (credential exposure) | scripts/persist-sold.js createClient | mitigate | Connection via `db.js createClient()` reading `.env` only; no secret literal; only counts logged, never row contents or credentials. |
| T-16-07 | Tampering (duplicate / data-loss on re-run) | ON CONFLICT DO UPDATE | mitigate | One row per stable key (booli_id / hemnet_slug); DO UPDATE refreshes columns from EXCLUDED without dropping the row. Re-running the persist pass converges (DB-03), never duplicates or destroys. |
| T-16-08 | Elevation / logic (title transfer in match set) | persistVerdictForRecord | mitigate | D-02 gate skips upsertSoldVerdict for `is_title_transfer` records (with config-predicate fallback); smoke asserts zero sold_match writes for a transfer record. |
