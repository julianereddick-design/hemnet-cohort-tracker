    Per the LOCKED summary keys in `<interfaces>` above:
    - Job A (hemnet-targeted-refresh) uses `rowsUpdated` + `rowsInserted` (NOT `updated`)
    - Job B (hemnet-targeted-match) uses `inserted` + `rowsUpdated`
    - Job C (booli-targeted-discovery) uses `inserted` + `updated`

    External-count clamp: `external_or_other = GREATEST(0, total - self_hosted)` with a `console.warn` whenever `self_hosted > total` (signals counter race or time-window mismatch — NOT real data anomaly; just informs the operator that the per-day numbers are slightly off and to look at the 24h-window edge cases).

    Write LITERALLY:

    ```js
    #!/usr/bin/env node
    // scripts/compare-writers.js — Phase 9 / SC-4 cutover observation.
    // Side-by-side row-count comparison: self-hosted (Job A/B/C, identified via
    // cron_job_log.result_summary using the LOCKED summary key names in
    // Plan 09-03 Task 1's <interfaces> block) vs. external scraper (the
    // residual — total 24h writes minus self-hosted writes, clamped at 0).
    //
    // Output format is deterministic so 7-day log scraping is trivial:
    //   DAY=<YYYY-MM-DD> TABLE=hemnet_listingv2 self_hosted_inserts=<N>
    //     self_hosted_updates=<N> external_or_other=<N> total_24h=<N>
    //   DAY=<YYYY-MM-DD> TABLE=booli_listing self_hosted_inserts=<N>
    //     self_hosted_updates=<N> external_or_other=<N> total_24h=<N>
    //   DAY=<YYYY-MM-DD> JOB=<name> status=<s> runs=<N>
    //
    // Locked summary keys (verified at Plan 09-03 PLAN time, do NOT re-discover):
    //   hemnet-targeted-refresh (Job A): rowsUpdated + rowsInserted (NO 'updated' key)
    //   hemnet-targeted-match   (Job B): inserted   + rowsUpdated   (UPSERT path)
    //   booli-targeted-discovery (Job C): inserted + updated         (simpler naming)
    //
    // External clamp: external_or_other = max(0, total - self_hosted). Warns when
    // self_hosted > total — that's a counter race or time-window edge, not real data.
    //
    // Exit 0 always (this is observational; alerting belongs to cron-wrapper Slack).

    'use strict';
    require('dotenv').config();
    const { createClient } = require('../db');

    function fmtDate(d) {
      return d.toISOString().slice(0, 10);
    }

    function clampExternal(total, selfHosted, table) {
      if (selfHosted > total) {
        console.warn(
          `WARN compare-writers: self_hosted=${selfHosted} > total_24h=${total} ` +
          `for ${table}; clamping external_or_other to 0. Likely a counter race ` +
          `(self-hosted cron job overlapped the 24h window edge) — not a data anomaly. ` +
          `Re-run after the next full hour for a cleaner snapshot.`,
        );
        return 0;
      }
      return total - selfHosted;
    }

    async function main() {
      const client = createClient();
      await client.connect();
      try {
        const now = new Date();
        const day = fmtDate(now);

        // === hemnet_listingv2 ===
        // Total 24h writes:
        const totalHemnetRes = await client.query(
          `SELECT COUNT(*)::int AS n FROM hemnet_listingv2
           WHERE created_at > NOW() - INTERVAL '24 hours'`,
        );
        const totalHemnet = totalHemnetRes.rows[0].n;

        // Self-hosted hemnet INSERTS: from hemnet-targeted-match (Job B's primary
        // insert path). Locked key: result_summary->>'inserted'.
        const hemnetMatchRes = await client.query(
          `SELECT COALESCE(SUM((result_summary->>'inserted')::int), 0)::int AS inserted
           FROM cron_job_log
           WHERE script_name = 'hemnet-targeted-match'
             AND status IN ('success','warning')
             AND started_at > NOW() - INTERVAL '24 hours'`,
        );
        const selfHostedHemnetInserted = hemnetMatchRes.rows[0].inserted;

        // Self-hosted hemnet UPDATES: from hemnet-targeted-refresh (Job A's primary
        // update path). Locked key: result_summary->>'rowsUpdated' (NOT 'updated').
        const hemnetRefreshRes = await client.query(
          `SELECT COALESCE(SUM((result_summary->>'rowsUpdated')::int), 0)::int AS rows_updated
           FROM cron_job_log
           WHERE script_name = 'hemnet-targeted-refresh'
             AND status IN ('success','warning')
             AND started_at > NOW() - INTERVAL '24 hours'`,
        );
        const selfHostedHemnetUpdated = hemnetRefreshRes.rows[0].rows_updated;

        // External clamp: only inserts count against total (updates don't change row count).
        const externalHemnet = clampExternal(totalHemnet, selfHostedHemnetInserted, 'hemnet_listingv2');
        console.log(
          `DAY=${day} TABLE=hemnet_listingv2 self_hosted_inserts=${selfHostedHemnetInserted} ` +
          `self_hosted_updates=${selfHostedHemnetUpdated} external_or_other=${externalHemnet} ` +
          `total_24h=${totalHemnet}`,
        );

        // === booli_listing ===
        const totalBooliRes = await client.query(
          `SELECT COUNT(*)::int AS n FROM booli_listing
           WHERE crawled > NOW() - INTERVAL '24 hours'`,
        );
        const totalBooli = totalBooliRes.rows[0].n;

        // booli-targeted-discovery (Job C) uses 'inserted' + 'updated' (simpler naming
        // than the hemnet-targeted-* scripts). Both bump 'crawled' per the ON CONFLICT
        // DO UPDATE clause in booli-targeted-discovery.js:200-213, so total_24h on
        // booli_listing.crawled includes both inserts AND updates from Job C.
        const booliDiscRes = await client.query(
          `SELECT COALESCE(SUM((result_summary->>'inserted')::int), 0)::int AS inserted,
                  COALESCE(SUM((result_summary->>'updated')::int), 0)::int  AS updated
           FROM cron_job_log
           WHERE script_name = 'booli-targeted-discovery'
             AND status IN ('success','warning')
             AND started_at > NOW() - INTERVAL '24 hours'`,
        );
        const selfHostedBooliInserted = booliDiscRes.rows[0].inserted;
        const selfHostedBooliUpdated = booliDiscRes.rows[0].updated;
        const selfHostedBooliCrawled = selfHostedBooliInserted + selfHostedBooliUpdated;
        const externalBooli = clampExternal(totalBooli, selfHostedBooliCrawled, 'booli_listing');
        console.log(
          `DAY=${day} TABLE=booli_listing self_hosted_inserts=${selfHostedBooliInserted} ` +
          `self_hosted_updates=${selfHostedBooliUpdated} external_or_other=${externalBooli} ` +
          `total_24h=${totalBooli}`,
        );

        // === Status summary ===
        const statusRes = await client.query(
          `SELECT script_name, status, COUNT(*)::int AS n
           FROM cron_job_log
           WHERE started_at > NOW() - INTERVAL '24 hours'
             AND script_name IN ('hemnet-targeted-refresh','hemnet-targeted-match','booli-targeted-discovery')
           GROUP BY script_name, status
           ORDER BY script_name, status`,
        );
        for (const r of statusRes.rows) {
          console.log(`DAY=${day} JOB=${r.script_name} status=${r.status} runs=${r.n}`);
        }
      } finally {
        await client.end();
      }
    }

    main().catch((e) => { console.error(e); process.exit(1); });
    ```
