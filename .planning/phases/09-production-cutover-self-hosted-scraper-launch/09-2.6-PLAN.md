2. **Worker loop check + outer try/catch** at the top of each iteration (around `:663-678`). Current worker body (verified via Read of hemnet-targeted-match.js:663-678) has NO outer try/catch — at conc 8 with Oxylabs Advanced under stress, an unforeseen throw at queue.shift() / jitter() / processOne level would silently reduce effective concurrency. Match `booli-targeted-refresh.js:287-321` exactly:

```js
async function worker() {
  while (queue.length) {
    if ((Date.now() - startMs) >= JOB_BUDGET_MS) {
      summary.budgetExceeded = true;
      log('WARN', `job-budget-exceeded ms=${Date.now() - startMs} remaining-queue=${queue.length} — draining`);
      queue.length = 0;
      break;
    }
    const row = queue.shift();
    if (row == null) break;
    try {
      await sleep(jitter());
      await processOne(row, client, log, dryRun, summary, searchCache, searchInFlight);
      processedCount++;
      if (processedCount % 25 === 0) {
        log(
          'INFO',
          `processed ${processedCount}/${booliRows.length} ` +
          `(matched: ${summary.matchedFromSearch}, errors: ${summary.fetchErrors + summary.parseErrors})`,
        );
      }
    } catch (workerErr) {
      const detail = String(
        (workerErr && workerErr.stack) ||
        (workerErr && workerErr.message) ||
        JSON.stringify(workerErr),
      );
      summary.workerErrors++;
      log('ERROR',
        `worker-uncaught booli_id=${row && row.booli_id} err=${detail}`,
      );
    }
  }
}
```

3. **Summary keys** — add BOTH `budgetExceeded: false` AND `workerErrors: 0` to the pre-allocated `summary` object around `:632-653` (mirror Job D — see `booli-targeted-refresh.js:280-285` for the workerErrors initialization pattern).

if (summary.budgetExceeded === true) {

}
if (summary.workerErrors > 0) {
