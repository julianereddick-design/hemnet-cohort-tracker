    **Step G: processOne worker body.** This is the Booli swap of hemnet-targeted-refresh.js:96-190. Write LITERALLY (with the Booli-specific SQL from D-03 and the defensive INSERT from D-04):
    ```js
    async function processOne(pair, client, log, dryRun, summary) {
      const { booli_id, url } = pair;
      let result;
      try {
        result = await fetchBooliDetail(url, { logger: log });
      } catch (err) {
        log('ERROR', `booli_id=${booli_id} url=${url} fetch error: ${err && err.message}`);
        summary.errors++;
        bucket(summary, '(unknown)').errors++;
        return; // leave existing booli_listing row UNTOUCHED
      }

      summary.fetched++;

      if (result.status === 'active') {
        const listing = result.listing;
        summary.parsed++;
        summary.activeCount++;
        const countyKey = (listing && listing.county) || '(unknown)';
        bucket(summary, countyKey).active++;

        if (!dryRun) {
          const shaped = shapeBooliForUpdate(listing);
          // UPDATE the booli_listing row for this booli_id (D-03).
          // days_listed is computed in SQL from (CURRENT_DATE - listed)::int so
          // there's no JS-side timezone risk; Job D never writes 'listed' (Job C owns it).
          const upd = await client.query(
            `UPDATE booli_listing
                SET times_viewed = $1,
                    is_active    = true,
                    crawled      = NOW(),
                    days_listed  = (CURRENT_DATE - listed)::int
              WHERE booli_id = $2
             RETURNING booli_id`,
            [shaped.times_viewed, booli_id],
          );
          if (upd.rowCount > 0) {
            summary.rowsUpdated += upd.rowCount;
          } else {
            // Defensive INSERT fallback (D-04). cohort_pairs.booli_id should always
            // have a backing booli_listing row (cohort-create.js guarantees this),
            // but the safety net mirrors Job A symmetry and costs ~15 lines.
            // Use to_timestamp on published for the 'listed' date column.
            // NOTE: this fallback writes a minimal row — title/street/county/muni
            // come from the EXCLUDED set Job C populates. If we hit this branch,
            // a subsequent Job C run will fill in the descriptive columns.
            try {
              await client.query(
                `INSERT INTO booli_listing
                   (id, url, booli_id, is_active, listed, crawled, times_viewed,
                    title, street_address, county, municipality, district,
                    postcode, currency, days_listed, is_pre_market)
                 VALUES
                   (nextval('booli_listing_id_seq'), $1, $2, true,
                    to_timestamp($3)::date, NOW(), $4,
                    '', '', '', '', '', NULL, 'SEK',
                    GREATEST(0, (CURRENT_DATE - to_timestamp($3)::date)::int), false)
                 ON CONFLICT (booli_id) DO NOTHING`,
                [url, booli_id, shaped.published_at_seconds, shaped.times_viewed],
              );
              summary.rowsInserted++;
            } catch (insErr) {
              log('ERROR',
                `defensive-insert-failed booli_id=${booli_id} url=${url}: ${insErr && insErr.message}`,
              );
              // Swallow — this is defense-in-depth, not the primary path.
            }
          }
        }
        return;
      }

      // result.status === 'inactive'
      if (result.reason === '404') {
        summary.removed404++;
      } else {
        summary.inactiveCount++;
      }
      bucket(summary, '(unknown)').inactive++;

      if (!dryRun) {
        // Preserve times_viewed; only flip is_active (D-03 — same defensive pattern
        // as Job A at hemnet-targeted-refresh.js:181-189).
        const upd = await client.query(
          `UPDATE booli_listing
              SET is_active = false
            WHERE booli_id = $1
           RETURNING booli_id`,
          [booli_id],
        );
        if (upd.rowCount > 0) summary.rowsUpdated += upd.rowCount;
      }
    }
    ```

This is belt-and-suspenders — cohort_pairs.booli_id should always have a backing row, but the constraint catches the impossible-but-cheap-to-handle case. If your inspection of booli_listing's actual UNIQUE constraints (CONFLICT target) finds the unique constraint is on `url` not `booli_id`, switch to `ON CONFLICT (url) DO NOTHING` — check booli-targeted-discovery.js:207 for the constraint Job C uses (UNIQUE(url) is the documented Phase 8.5 constraint per booli-targeted-discovery.js header line 21).
